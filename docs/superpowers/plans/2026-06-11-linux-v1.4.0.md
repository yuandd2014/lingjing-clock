// js/platform.js — 跨平台行为抽象层
// 避免在 main.js 散落 process.platform === 'win32' 判断
const { app, screen } = require('electron');
const fs = require('fs');
const path = require('path');
const { exec, execSync } = require('child_process');

const platform = process.platform;  // 'win32' | 'linux' | 'darwin'

function isWin() { return platform === 'win32'; }
function isLinux() { return platform === 'linux'; }
function isMac() { return platform === 'darwin'; }

// === 自启动 (Windows 注册表 / Linux .desktop) ===
function setAutoLaunch(enabled) {
  if (isWin()) {
    // Windows: 走 app.setLoginItemSettings (跟 NSIS 写注册表等效)
    app.setLoginItemSettings({
      openAtLogin: enabled,
      path: process.execPath,
      args: []
    });
  } else if (isLinux()) {
    // Linux: 写 ~/.config/autostart/lingjing-clock.desktop
    const home = app.getPath('home');
    const autostartDir = path.join(home, '.config', 'autostart');
    const desktopFile = path.join(autostartDir, 'lingjing-clock.desktop');
    if (enabled) {
      fs.mkdirSync(autostartDir, { recursive: true });
      const content = `[Desktop Entry]
Type=Application
Name=灵境时钟
Exec=${process.execPath}
Icon=lingjing-clock
X-GNOME-Autostart-enabled=true
NoDisplay=false
Terminal=false
`;
      fs.writeFileSync(desktopFile, content, { encoding: 'utf8' });
      fs.chmodSync(desktopFile, 0o755);
    } else {
      try { fs.unlinkSync(desktopFile); } catch (e) { /* 文件不存在 */ }
    }
  }
}

function getAutoLaunch() {
  if (isWin()) {
    return app.getLoginItemSettings().openAtLogin;
  } else if (isLinux()) {
    const home = app.getPath('home');
    const desktopFile = path.join(home, '.config', 'autostart', 'lingjing-clock.desktop');
    return fs.existsSync(desktopFile);
  }
  return false;
}

// === 单实例锁 ===
let gotLock = null;
function requestSingleInstance(handler) {
  if (isWin()) {
    gotLock = app.requestSingleInstanceLock();
    if (!gotLock) {
      app.quit();
      return false;
    }
    app.on('second-instance', handler);
    return true;
  } else if (isLinux()) {
    // Linux: 走 lockfile (DBus 在某些 DE 上不稳)
    const lockDir = require('os').tmpdir();
    const lockFile = path.join(lockDir, 'lingjing-clock.lock');
    try {
      if (fs.existsSync(lockFile)) {
        const pid = parseInt(fs.readFileSync(lockFile, 'utf8').trim(), 10);
        if (pid && process.kill(pid, 0)) {
          // 进程存活 → 拒绝启动
          app.quit();
          return false;
        }
      }
      fs.writeFileSync(lockFile, String(process.pid));
      app.on('will-quit', () => {
        try { fs.unlinkSync(lockFile); } catch (e) {}
      });
      app.on('second-instance', handler);
      return true;
    } catch (e) {
      // lockfile 失败 → 允许启动 (不影响主功能)
      return true;
    }
  }
  return true;
}

// === 自动更新启用状态 (Windows 注册表 / Linux deb conf + settings) ===
function readAutoUpdateEnabled() {
  // 1. 优先读系统级开关
  if (isWin()) {
    try {
      const out = execSync('reg query "HKCU\\Software\\LingJing Clock" /v AutoUpdate 2>nul',
        { encoding: 'utf8', windowsHide: true });
      if (/AutoUpdate\s+REG_DWORD\s+0x0/.test(out)) return false;
      if (/AutoUpdate\s+REG_DWORD\s+0x1/.test(out)) return true;
    } catch (e) {}
  } else if (isLinux()) {
    // Linux: 读 ~/.config/LingJing Clock/lingjing-settings-v1.json (跟 Windows settings 共用)
    // 暂不读 deb conffile (debian/conf 文件需要特殊 hook)
  }
  // 2. 读 app 内设置 (跨平台统一)
  try {
    const userData = app.getPath('userData');
    const settingsPath = path.join(userData, 'lingjing-settings-v1.json');
    if (fs.existsSync(settingsPath)) {
      const s = JSON.parse(fs.readFileSync(settingsPath, 'utf8'));
      if (s && typeof s.autoUpdateEnabled === 'boolean') return s.autoUpdateEnabled;
    }
  } catch (e) {}
  return true;  // 默认开启
}

module.exports = {
  platform, isWin, isLinux, isMac,
  setAutoLaunch, getAutoLaunch,
  requestSingleInstance,
  readAutoUpdateEnabled,
};
```

// js/tray.js — 跨平台系统托盘
const { Tray, Menu, nativeImage, app } = require('electron');
const path = require('path');
const fs = require('fs');
const { isWin, isLinux } = require('./platform.js');

let trayInstance = null;

function createTray(getMainWindow, onShow, onHide, onQuit, onReload) {
  let iconPath;
  if (isWin()) {
    iconPath = path.join(__dirname, '..', 'build', 'installerHeaderIcon.ico');
  } else if (isLinux()) {
    iconPath = path.join(__dirname, '..', 'build', 'linux', 'icon-32.png');
  }
  if (!fs.existsSync(iconPath)) {
    console.warn('[tray] icon not found:', iconPath);
    return null;
  }
  const icon = nativeImage.createFromPath(iconPath);
  trayInstance = new Tray(icon);
  trayInstance.setToolTip('灵境时钟 · LingJing Clock');

  rebuildMenu(getMainWindow, onShow, onHide, onQuit, onReload);

  trayInstance.on('click', () => {
    // Windows: click 弹菜单; Linux: click 通常是 toggle 窗口
    const win = getMainWindow();
    if (win && !win.isDestroyed()) {
      if (win.isVisible()) win.hide();
      else { win.show(); win.focus(); }
    }
  });

  return trayInstance;
}

function rebuildMenu(getMainWindow, onShow, onHide, onQuit, onReload) {
  if (!trayInstance) return;
  const win = getMainWindow();
  const contextMenu = Menu.buildFromTemplate([
    {
      label: '显示主窗口',
      click: () => onShow && onShow(),
      enabled: !!(win && win.isVisible())
    },
    {
      label: '隐藏主窗口',
      click: () => onHide && onHide(),
      enabled: !!(win && !win.isVisible())
    },
    { type: 'separator' },
    {
      label: '重新加载 (Ctrl+R)',
      click: () => onReload && onReload()
    },
    { type: 'separator' },
    {
      label: '退出',
      click: () => onQuit && onQuit()
    },
  ]);
  trayInstance.setContextMenu(contextMenu);
}

function destroyTray() {
  if (trayInstance && !trayInstance.isDestroyed()) {
    trayInstance.destroy();
  }
  trayInstance = null;
}

module.exports = { createTray, rebuildMenu, destroyTray };
```

1. **CI 流水线第一跑失败** (electron-builder Linux 依赖 / 镜像问题) — 备选: 跑 `xvfb-run electron-builder --linux` 在 ubuntu-latest
2. **WSL2 缺桌面环境** — 用户需提供 (WSLg 已经默认有 Wayland)
3. **Wayland 透明背景** — Electron 31 自动降级为不透明, 不影响功能
4. **libappindicator 缺失** — deb depends 已经包含, AppImage 用户需 `apt install libappindicator3-1`
5. **Linux 自动更新** — v1.4.0 先不上, 后续 v1.4.x PATCH 走 AppImageUpdate
